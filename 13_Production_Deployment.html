Basic FastAPI App — Beginner Example

Python — main.py 
from fastapi import FastAPI
from pydantic import BaseModel
from typing import Optional

# FastAPI app create karo
app = FastAPI(
    title="AI Agent API",
    version="1.0.0",
    description="Mera pehla AI backend!"
)

# Pydantic model — Request body ka structure
class ChatRequest(BaseModel):
    message: str
    user_id: Optional[str] = "anonymous"

# Response model
class ChatResponse(BaseModel):
    reply: str
    tokens_used: int

# GET endpoint — Simple health check
@app.get("/")
async def root():
    return {"status": "✅ Server chal raha hai!"}

# POST endpoint — Chat message receive karo
@app.post("/chat", response_model=ChatResponse)
async def chat(request: ChatRequest):
    # Yahan AI model call hoga (abhi simple response)
    reply = f"Aapne kaha: {request.message}"
    return ChatResponse(
        reply=reply,
        tokens_used=len(request.message.split())
    )

# Run command:
# uvicorn main:app --reload --host 0.0.0.0 --port 8000

Real-World FastAPI AI App — Claude Integration

Python — ai_server.py 
from fastapi import FastAPI, HTTPException, Depends
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
import anthropic
import os
from dotenv import load_dotenv

load_dotenv()  # .env file se API keys load karo

app = FastAPI(title="AI Chat Server")

# CORS allow karo (frontend se requests aane ke liye)
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # Production mein specific domain dena
    allow_methods=["*"],
    allow_headers=["*"],
)

class Message(BaseModel):
    role: str      # "user" ya "assistant"
    content: str

class ChatReq(BaseModel):
    messages: list[Message]
    system: str = "You are a helpful AI assistant."

@app.post("/api/chat")
async def chat_endpoint(req: ChatReq):
    try:
        client = anthropic.Anthropic(
            api_key=os.getenv("ANTHROPIC_API_KEY")
        )
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=1024,
            system=req.system,
            messages=[m.dict() for m in req.messages]
        )
        return {
            "reply": response.content[0].text,
            "usage": response.usage.model_dump()
        }
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
✅ FastAPI Best Practice
Hamesha async def use karo endpoints ke liye jab bhi I/O operations ho (database, external API). Isse server ek request ke wait mein dusre requests serve karta rehta hai.

Python — Flask AI Server 
from flask import Flask, request, jsonify
from flask_cors import CORS
import anthropic, os

app = Flask(__name__)
CORS(app)  # Cross-origin requests allow karo

client = anthropic.Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))

@app.route("/health", methods=["GET"])
def health():
    return jsonify({"status": "ok"})

@app.route("/summarize", methods=["POST"])
def summarize():
    data = request.get_json()
    text = data.get("text", "")
    
    if not text:
        return jsonify({"error": "text field chahiye!"}), 400
    
    resp = client.messages.create(
        model="claude-haiku-4-5-20251001",  # Faster, cheaper model
        max_tokens=500,
        messages=[{
            "role": "user",
            "content": f"Yeh text summarize karo:\n\n{text}"
        }]
    )
    return jsonify({"summary": resp.content[0].text})

if __name__ == "__main__":
    app.run(debug=False, host="0.0.0.0", port=5000)

const express = require('express');
const Anthropic = require('@anthropic-ai/sdk');

const app = express();
app.use(express.json());

const client = new Anthropic();

// Streaming response — real-time text generate karo
app.post('/stream', async (req, res) => {
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    
    const stream = client.messages.stream({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1024,
        messages: [{ role: 'user', content: req.body.message }]
    });
    
    // Har text chunk stream karo client ko
    stream.on('text', (text) => {
        res.write(`data: ${JSON.stringify({text})}\n\n`);
    });
    
    stream.on('finalMessage', () => {
        res.write('data: [DONE]\n\n');
        res.end();
    });
});

app.listen(3000, () => console.log('🚀 Server port 3000 par chal raha hai!'));

Python — AWS Lambda Function (AI) 
import json
import anthropic
import os

# Lambda handler function — yeh AWS call karta hai
def handler(event, context):
    """
    event: HTTP request data (body, headers, etc.)
    context: Lambda runtime information
    """
    try:
        # Request body parse karo
        body = json.loads(event.get('body', '{}'))
        message = body.get('message', '')
        
        if not message:
            return {
                'statusCode': 400,
                'body': json.dumps({'error': 'message required'})
            }
        
        # Claude API call
        client = anthropic.Anthropic(
            api_key=os.environ['ANTHROPIC_API_KEY']  # Env var se lo
        )
        resp = client.messages.create(
            model='claude-haiku-4-5-20251001',
            max_tokens=256,
            messages=[{'role': 'user', 'content': message}]
        )
        
        return {
            'statusCode': 200,
            'headers': {'Content-Type': 'application/json'},
            'body': json.dumps({
                'reply': resp.content[0].text
            })
        }
    except Exception as e:
        return {'statusCode': 500, 'body': json.dumps({'error': str(e)})}

4.1 Load Balancing

Load Balancer Kya Hai?

Load Balancer ek traffic cop hai jo incoming requests ko multiple servers mein distribute karta hai taaki koi ek server overloaded na ho.

Load Balancer — Traffic Distribute Karna

Load Balancing Algorithms

AlgorithmKaise Kaam Karta HaiBest For
Round RobinServer 1, 2, 3, 1, 2, 3... (rotate karo)Equal servers, simple cases
Least ConnectionsSabse kam busy server ko bhejoVariable processing time (AI calls!)
IP HashUser ka IP dekh ke same server pe bhejoSession persistence (login apps)
WeightedPowerful servers ko zyada trafficMixed hardware setups
RandomRandom server choose karoSimple stateless services
✅ AI App ke Liye Best Algorithm
AI apps ke liye Least Connections use karo. Kyunki Claude/GPT calls ka response time variable hota hai — koi query 2 sec mein done, koi 10 sec mein. Least connections algorithm smartly route karta hai.

4.2 Caching — Repeat Kaam Mat Karo

Caching Kya Hai?

Caching matlab hai pehle se compute kiye results store karna taaki baar baar wahi kaam na karna pade. AI apps mein yeh bahut important hai kyunki LLM calls expensive ($) aur slow (seconds) hote hain.

Python — Redis Caching with FastAPI 
import redis
import hashlib
import json
from fastapi import FastAPI

app = FastAPI()
cache = redis.Redis(host='localhost', port=6379)

def get_cache_key(text: str) -> str:
    # Text ka unique hash banao (same text = same key)
    return f"ai:response:{hashlib.md5(text.encode()).hexdigest()}"

@app.post("/summarize")
async def summarize(text: str):
    key = get_cache_key(text)
    
    # Step 1: Cache check karo pehle
    cached = cache.get(key)
    if cached:
        return {"summary": json.loads(cached), "from_cache": True}
    
    # Step 2: Cache miss — AI call karo
    result = await call_claude_api(text)
    
    # Step 3: Result cache mein save karo (1 hour ke liye)
    cache.setex(key, 3600, json.dumps(result))
    
    return {"summary": result, "from_cache": False}
    
# Cache hit rate 70%+ → Cost 70% kam! 🎉

from fastapi import FastAPI, HTTPException, Depends
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
import jwt
from datetime import datetime, timedelta
import os

SECRET_KEY = os.getenv("JWT_SECRET")  # Strong secret key chahiye!
security = HTTPBearer()

def create_token(user_id: str) -> str:
    payload = {
        "user_id": user_id,
        "exp": datetime.utcnow() + timedelta(hours=24)  # 24h expiry
    }
    return jwt.encode(payload, SECRET_KEY, algorithm="HS256")

def verify_token(creds: HTTPAuthorizationCredentials = Depends(security)):
    try:
        payload = jwt.decode(creds.credentials, SECRET_KEY, algorithms=["HS256"])
        return payload
    except jwt.ExpiredSignatureError:
        raise HTTPException(401, "Token expired! Dobara login karo.")
    except jwt.InvalidTokenError:
        raise HTTPException(401, "Invalid token!")

5.2 Rate Limiting — Abuse Rokna

Rate limiting matlab ek user kitni requests per minute kar sakta hai — yeh limit set karna. AI apps ke liye critical hai kyunki bina limit ke koi ek user thousands of dollars ka API cost kar sakta hai!

Python — Rate Limiting with Redis 
import redis
from fastapi import HTTPException, Request
from functools import wraps

redis_client = redis.Redis(host='localhost')

def rate_limit(max_requests: int = 10, window_seconds: int = 60):
    """10 requests per 60 seconds per user"""
    def decorator(func):
        @wraps(func)
        async def wrapper(request: Request, *args, **kwargs):
            # User ka IP ya user_id se key banao
            client_ip = request.client.host
            key = f"rate:{client_ip}"
            
            # Current count check karo
            count = redis_client.get(key)
            if count and int(count) >= max_requests:
                raise HTTPException(
                    429,
                    f"Bahut zyada requests! {window_seconds}s baad try karo."
                )
            
            # Count increment karo
            redis_client.incr(key)
            redis_client.expire(key, window_seconds)  # Auto expire
            
            return await func(request, *args, **kwargs)
        return wrapper
    return decorator

# Use karna aasaan hai:
@app.post("/chat")
@rate_limit(max_requests=10, window_seconds=60)
async def chat(request: Request, ...):
    ...

def sanitize_user_input(user_message: str) -> str:
    """
    Prompt injection attacks filter karo:
    - "Ignore all instructions" jaisi strings
    - System prompt override attempts
    - Jailbreak patterns

    if len(user_message) > 10000:
        raise ValueError("Input too long!")
    
    return user_message.strip()

Python — Smart Model Router 
import anthropic

def route_to_model(query: str) -> str:
    """Query ki complexity dekh ke best model choose karo"""
    query_lower = query.lower()
    word_count = len(query.split())
    
    # Simple queries → Haiku (cheapest, fastest)
    simple_keywords = ["kya hai", "what is", "define", "translate"]
    if word_count < 20 or any(k in query_lower for k in simple_keywords):
        return "claude-haiku-4-5-20251001"  # 60x cheaper!
    
    # Complex queries → Opus (expensive but best)
    complex_keywords = ["analyze", "research", "strategy", "compare"]
    if word_count > 100 or any(k in query_lower for k in complex_keywords):
        return "claude-opus-4-6"
    
    # Default → Sonnet (balanced)
    return "claude-sonnet-4-6"

async def smart_chat(user_query: str):
    model = route_to_model(user_query)
    client = anthropic.Anthropic()
    
    response = client.messages.create(
        model=model,
        max_tokens=1024,
        messages=[{"role": "user", "content": user_query}]
    )
    return {"reply": response.content[0].text, "model_used": model}
