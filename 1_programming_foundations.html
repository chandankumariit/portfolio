# Generator — large document processing ke liye
def chunk_text(text, size=512):
    """Text ko chunks mein yield karo — memory efficient"""
    for i in range(0, len(text), size):
        yield text[i:i+size]

1.4.4 Recursion

📖 Definition
Recursion tab hoti hai jab ek function khud ko call karta hai. Har recursive call ek smaller problem solve karta hai. Ek base case zaroori hai — warna infinite loop. 
# Recursive agent — task decompose karo sub-tasks mein
def decompose_task(task: str, depth: int = 0, max_depth: int = 3):
    indent = "  " * depth
    print(f"{indent}Task: {task}")
    if depth >= max_depth:              # Base case — ruk jao
        print(f"{indent}→ Execute directly")
        return
    # Sub-tasks (simplified for demo)
    subtasks = [f"{task}_step{i}" for i in range(1, 3)]
    for sub in subtasks:
        decompose_task(sub, depth + 1, max_depth)  # Recursive call

decompose_task("Research AI agents")

Closure

Decorator

Function ko wrap karta hai — bina original function change kiye behavior add karna. Logging, retry, auth mein use.

def log_call(func):
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        result = func(*args, **kwargs)
        print(f"Done: {func.__name__}")
        return result
    return wrapper

@log_call
def search_web(query):
    return f"Results for {query}"

search_web("RAG")
Python
⚡ Production Decorator — Retry Logic
import time
def retry(max_attempts=3, delay=1):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try: return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1: raise
                    time.sleep(delay * (2**attempt))  # Exponential backoff
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2)
def call_llm_api(prompt): pass  # Real API call yahan aayega
Python

class Agent:
    """AI Agent base class"""

    # Class variable (shared by all instances)
    agent_count = 0

    def __init__(self, name: str, tools: list):
        """Constructor — object create hote waqt chalta hai"""
        self.name = name                 # Instance variable
        self.tools = tools
        self.memory = []
        self.iteration = 0
        Agent.agent_count += 1          # Class variable update

    def think(self, query: str) -> str:
        """LLM se reasoning"""
        self.memory.append({"thought": query})
        return f"[{self.name}] Thinking about: {query}"

    def act(self, tool_name: str, *args) -> str:
        if tool_name not in self.tools:
            return f"Error: Tool {tool_name} not available"
        return f"[{self.name}] Using {tool_name} with {args}"

    def __repr__(self) -> str:
        """Object ka string representation"""
        return f"Agent(name={self.name!r}, tools={self.tools})"

# Object creation
bot = Agent("ResearchBot", ["search", "summarize"])
print(bot.think("What is RAG?"))
print(bot.act("search", "RAG papers 2024"))
print(Agent.agent_count)  # 1
Python

1.5.2 Inheritance — Parent se Sikhna

class ResearchAgent(Agent):   # Agent se inherit kar raha hai
    def __init__(self, name: str):
        super().__init__(name, tools=["search", "arxiv", "summarize"])
        self.papers_found = []

    def research(self, topic: str) -> str:
        self.think(f"Researching: {topic}")      # Parent method
        result = self.act("search", topic)        # Parent method
        self.papers_found.append(topic)
        return result

class CodingAgent(Agent):
    def __init__(self, name: str):
        super().__init__(name, tools=["code_interpreter", "debugger"])

    def code(self, task: str) -> str:

import requests
import logging

def call_llm_api(prompt: str, timeout: int = 30) -> dict:
    """Robust LLM API call with proper error handling"""
    try:
        response = requests.post(
            "https://api.openai.com/v1/chat/completions",
            json={"model": "gpt-4", "messages": [{"role":"user","content":prompt}]},
            timeout=timeout
        )
        response.raise_for_status()       # 4xx/5xx pe exception raise karo
        return response.json()

    except requests.exceptions.Timeout:
        logging.error("API timeout after %ds", timeout)
        return {"error": "timeout", "retry": True}

    except requests.exceptions.HTTPError as e:
        if e.response.status_code == 429:   # Rate limit
            return {"error": "rate_limit", "retry": True}
        return {"error": str(e), "retry": False}

    except Exception as e:
        logging.exception("Unexpected error in LLM call")
        return {"error": str(e), "retry": False}

    finally:
        logging.info("LLM API call completed")    # Hamesha log karo

"Sahi data structure = fast code. Galat data structure = production mein rona."

1.8.1 Core Data Structures

Array / List — O(1) access

Index pe direct access. Insert/delete at end: O(1). Middle mein: O(n). AI use: token sequences, retrieved chunks.

Stack (LIFO) — O(1) push/pop

Last In, First Out. Function call stack, undo operations, DFS traversal. Python mein list use karo.

Queue (FIFO) — O(1) enqueue/dequeue

First In, First Out. Task queues, BFS, message queues. collections.deque use karo.

HashMap / Dict — O(1) avg lookup

Key-value pairs. Tool registry, cache, memoization. Python dict hash-table backed hai.

Tree — O(log n) operations

Hierarchical data. File systems, decision trees, task decomposition trees.

Graph — O(V+E) traversal

Nodes aur edges. Knowledge graphs, multi-agent communication, LangGraph.

1.8.3 Searching

# Linear Search — O(n)
def linear_search(arr, target):
    for i, val in enumerate(arr):
        if val == target:
            return i
    return -1

# Binary Search — O(log n) — sorted array pe
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target: return mid
        elif arr[mid] < target: left = mid + 1
        else: right = mid - 1
    return -1

import requests

def chat_completion(messages: list, model: str = "gpt-4o") -> str:
    """OpenAI Chat Completion API call"""

import sqlite3
from contextlib import contextmanager

@contextmanager
def get_db(path="agent.db"):
    conn = sqlite3.connect(path)
    conn.row_factory = sqlite3.Row   # dict-like rows
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

import asyncio
import aiohttp

# Async function — 'async def' se define hoti hai
async def fetch_tool_result(session: aiohttp.ClientSession, tool_url: str) -> dict:
    """Single tool API call"""
    async with session.get(tool_url) as response:
        return await response.json()   # await se wait karo

async def run_parallel_tools(tool_urls: list) -> list:
    """Multiple tools concurrently chalao"""
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_tool_result(session, url) for url in tool_urls]
        results = await asyncio.gather(*tasks)  # Sab ek saath
        return results

# Synchronous context se async function call karna
urls = ["https://api.tool1.com", "https://api.tool2.com"]
results = asyncio.run(run_parallel_tools(urls))

            tool = "echo"  # Real: LLM decides
            if tool in self.tools:
                result = self.tools[tool](query)
            else:
                result = f"Processed: {query}"
            self.history.append({"role": "assistant", "content": result})
            return result
        except Exception as e:
            logging.error("Agent error: %s", e)
            return f"Error: {e}"

    def save_history(self, path: str = "history.json"):
        with open(path, "w") as f:
            json.dump(self.history, f, indent=2)

async def main():
    agent = SimpleAgent("MyFirstAgent")
    agent.register_tool("echo", lambda q: f"Echo: {q}")
    result = await agent.run("Hello, Agent!")
    print(result)
    agent.save_history()

asyncio.run(main())
